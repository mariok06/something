#include <stdio.h>

int n, target, arr[20], subset[20];

void sumOfSubsets(int i, int sum) {
    if (sum == target) {
        printf("Subset: ");
        for (int j = 0; j < i; j++)
            if (subset[j]) printf("%d ", arr[j]);
        printf("\n");
        return;
    }
    if (i == n || sum > target) return;

    subset[i] = 1; // include arr[i]
    sumOfSubsets(i + 1, sum + arr[i]);

    subset[i] = 0; // exclude arr[i]
    sumOfSubsets(i + 1, sum);
}

int main() {
    printf("Enter number of elements: ");
    scanf("%d", &n);
    printf("Enter elements: ");
    for (int i = 0; i < n; i++)
        scanf("%d", &arr[i]);
    printf("Enter target sum: ");
    scanf("%d", &target);

    printf("\nSubsets with sum %d:\n", target);
    sumOfSubsets(0, 0);
    return 0;
}
