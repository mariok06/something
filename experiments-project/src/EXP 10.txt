#include <stdio.h>
#include <string.h>

// Function to print the entered text
void printText(char txt[]) {
    printf("Entered text: %s\n", txt);
}

// Function to search for a pattern inside text
void search(char txt[], char pat[]) {
    int i, j;
    int found = 0;
    int n = strlen(txt);
    int m = strlen(pat);

    // Remove newline from fgets input
    if (txt[n - 1] == '\n') txt[n - 1] = '\0';
    if (pat[m - 1] == '\n') pat[m - 1] = '\0';

    n = strlen(txt);
    m = strlen(pat);

    // Search pattern in text
    for (i = 0; i <= n - m; i++) {
        for (j = 0; j < m; j++) {
            if (txt[i + j] != pat[j])
                break;
        }

        if (j == m) {  // if pattern matched
            printf("Pattern found at index %d\n", i);
            found = 1;
        }
    }

    if (!found)
        printf("Pattern not found\n");
}

int main() {
    char txt[100], pat[50];

    printf("Enter the text (e.g., aaabbbb): ");
    fgets(txt, sizeof(txt), stdin);

    printf("Enter the pattern (e.g., bb): ");
    fgets(pat, sizeof(pat), stdin);

    printText(txt);
    search(txt, pat);

    return 0;
}
